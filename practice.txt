class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        
		int sum;
		int vector<int> wealht;

		for(int i = 0; i < accounts.size(); i++){
			for(int j = 0; j < accounts[i].size(); j++){
				sum += accounts[i][j];
			}
			wealht.push_back(sum);
			sum = 0;
		}

		return *max_element(wealht.begin(), wealht.end());
    }
};
